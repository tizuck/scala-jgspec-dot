225   implicit def nodesDecoder[T1](implicit t1Decoder:Decoder[T1]):Decoder[Nodes[T1]] = {
226     (c: HCursor) => {
227       val keys = c.keys.getOrElse(Nil)
228 
229       val nodes = for {key <- keys} yield {
230         for {
231           node <- c.downField(key).as[Node[T1]]
232         } yield {
233           node.copy(jsonkey = key)
234         }
235       }
236 
237       nodes.foldLeft(Right[DecodingFailure, Nodes[T1]](Nodes(List.empty[Node[T1]])).withLeft) {
238         case (acc, value) =>
239           acc match {
240             case l@Left(_) => l
241             case Right(v) =>
242               value match {
243                 case l@Left(_) => Left(DecodingFailure(l.value.reason, c.history))
244                 case Right(value) => Right(v.copy(nodes = v.nodes ++ List(value)))
245               }
246           }
247       }
